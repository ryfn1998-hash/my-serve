async function triggerGitHubAction() { const status = document.getElementById('status'); const btn = document.getElementById('start-btn'); // 填入你的資料 const GITHUB_TOKEN = "你的ghp_Token"; const REPO_OWNER = "ryfn1998-hash"; const REPO_NAME = "my-serve"; status.innerText = "正在呼叫 GitHub 後台..."; btn.disabled = true; try { const response = await fetch(`https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/dispatches`, { method: 'POST', headers: { 'Authorization': `token ${GITHUB_TOKEN}`, 'Accept': 'application/vnd.github.v3+json' }, body: JSON.stringify({ event_type: "launch-server" }) }); if (response.status === 204) { status.innerText = "指令已送達！請稍候 1 分鐘開服。"; status.style.color = "#55FF55"; } else if (response.status === 401) { status.innerText = "錯誤：Token 無效或過期"; status.style.color = "#FF5555"; btn.disabled = false; } else if (response.status === 404) { status.innerText = "錯誤：找不到倉庫或名稱填錯"; status.style.color = "#FF5555"; btn.disabled = false; } else { status.innerText = "失敗：代碼 " + response.status; btn.disabled = false; } } catch (e) { status.innerText = "連線失敗，請檢查網路。"; btn.disabled = false; } }
